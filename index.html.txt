        function runAudit() {
            const rawHtml = document.getElementById('htmlInput').value.trim();
            if (!rawHtml) {
                alert('कृपया पहले HTML कोड पेस्ट करें!');
                return;
            }

            // HTML parsing using DOMParser
            const parser = new DOMParser();
            const doc = parser.parseFromString(rawHtml, 'text/html');

            let totalScore = 0;
            let maxScore = 100;
            const checks = [];

            // 1. Meta Title Check
            const title = doc.querySelector('title');
            const titleText = title ? title.textContent.trim() : '';
            if (titleText.length >= 30 && titleText.length <= 60) {
                totalScore += 20;
                checks.push({ title: 'Meta Title', status: 'pass', text: `परफेक्ट टाइटल! (${titleText.length} वर्ण)`, detail: titleText });
            } else if (titleText.length > 0) {
                totalScore += 10;
                checks.push({ title: 'Meta Title', status: 'warn', text: `टाइटल की लंबाई 30-60 वर्णों के बीच होनी चाहिए। (अभी: ${titleText.length})`, detail: titleText });
            } else {
                checks.push({ title: 'Meta Title', status: 'fail', text: '<title> टैग नहीं मिला या खाली है!', detail: 'अनुपलब्ध' });
            }

            // 2. Meta Description Check
            const descMeta = doc.querySelector('meta[name="description"]');
            const descText = descMeta ? descMeta.getAttribute('content') || '' : '';
            if (descText.length >= 120 && descText.length <= 160) {
                totalScore += 20;
                checks.push({ title: 'Meta Description', status: 'pass', text: `उत्कृष्ट विवरण! (${descText.length} वर्ण)`, detail: descText });
            } else if (descText.length > 0) {
                totalScore += 10;
                checks.push({ title: 'Meta Description', status: 'warn', text: `विवरण 120-160 वर्णों का होना चाहिए। (अभी: ${descText.length})`, detail: descText });
            } else {
                checks.push({ title: 'Meta Description', status: 'fail', text: 'Meta Description अनुपलब्ध है!', detail: 'अनुपलब्ध' });
            }

            // 3. H1 Heading Check
            const h1Tags = doc.querySelectorAll('h1');
            if (h1Tags.length === 1) {
                totalScore += 20;
                checks.push({ title: 'H1 Heading', status: 'pass', text: 'पेज पर बिल्कुल सही 1 H1 टैग मौजूद है।', detail: h1Tags[0].textContent.trim() });
            } else if (h1Tags.length > 1) {
                totalScore += 10;
                checks.push({ title: 'H1 Heading', status: 'warn', text: `पेज पर ${h1Tags.length} H1 टैग्स हैं। केवल 1 H1 टैग होना चाहिए।`, detail: `${h1Tags.length} H1 tags found` });
            } else {
                checks.push({ title: 'H1 Heading', status: 'fail', text: 'पेज पर कोई H1 हेडिंग नहीं मिली!', detail: 'अनुपलब्ध' });
            }

            // 4. Image Alt Attribute Check
            const images = doc.querySelectorAll('img');
            let missingAlt = 0;
            images.forEach(img => {
                if (!img.hasAttribute('alt') || img.getAttribute('alt').trim() === '') {
                    missingAlt++;
                }
            });

            if (images.length === 0) {
                totalScore += 15;
                checks.push({ title: 'Image Alt Tags', status: 'pass', text: 'पेज पर कोई इमेज नहीं है।', detail: 'N/A' });
            } else if (missingAlt === 0) {
                totalScore += 15;
                checks.push({ title: 'Image Alt Tags', status: 'pass', text: `सभी ${images.length} इमेजेस में Alt टैग्स मौजूद हैं!`, detail: `100% Alt Coverage (${images.length}/${images.length})` });
            } else {
                checks.push({ title: 'Image Alt Tags', status: 'warn', text: `${missingAlt}/${images.length} इमेजेस में Alt टैग नहीं हैं।`, detail: `Missing Alt: ${missingAlt}` });
            }

            // 5. Viewport Meta Check (Mobile Friendliness)
            const viewport = doc.querySelector('meta[name="viewport"]');
            if (viewport) {
                totalScore += 15;
                checks.push({ title: 'Mobile Viewport', status: 'pass', text: 'मोबाइल-फ्रेंडली Viewport meta टैग मौजूद है।', detail: viewport.getAttribute('content') });
            } else {
                checks.push({ title: 'Mobile Viewport', status: 'fail', text: 'Viewport meta टैग गायब है! आपकी साइट मोबाइल फ्रेंडली नहीं दिखेगी।', detail: 'अनुपलब्ध' });
            }

            // 6. Word Count Check
            const textContent = doc.body ? doc.body.textContent.replace(/\s+/g, ' ').trim() : '';
            const wordCount = textContent ? textContent.split(' ').length : 0;
            if (wordCount >= 300) {
                totalScore += 10;
                checks.push({ title: 'Content Word Count', status: 'pass', text: `पर्याप्त कंटेंट! कुल शब्द: ${wordCount}`, detail: `Total Words: ${wordCount}` });
            } else {
                totalScore += 5;
                checks.push({ title: 'Content Word Count', status: 'warn', text: `कंटेंट कम है (${wordCount} शब्द)। कम से कम 300 शब्द अनुशंसित हैं।`, detail: `Total Words: ${wordCount}` });
            }

            // Render Results
            renderResults(totalScore, checks);
        }

        function renderResults(score, checks) {
            const resultsContainer = document.getElementById('results');
            const scoreCircle = document.getElementById('scoreCircle');
            const scoreSummary = document.getElementById('scoreSummary');
            const statsGrid = document.getElementById('statsGrid');

            // Show results container
            resultsContainer.style.display = 'flex';

            // Animate Score
            scoreCircle.textContent = score + '%';
            if (score >= 80) {
                scoreCircle.style.borderColor = 'var(--success)';
                scoreCircle.style.color = 'var(--success)';
                scoreSummary.textContent = 'शानदार! आपकी वेबसाइट का SEO काफी मजबूत है।';
            } else if (score >= 50) {
                scoreCircle.style.borderColor = 'var(--warning)';
                scoreCircle.style.color = 'var(--warning)';
                scoreSummary.textContent = 'अच्छा है, लेकिन कुछ सुधारों की आवश्यकता है।';
            } else {
                scoreCircle.style.borderColor = 'var(--danger)';
                scoreCircle.style.color = 'var(--danger)';
                scoreSummary.textContent = 'SEO स्कोर कम है! नीचे दिए सुझावों को ठीक करें।';
            }

            // Render Individual Cards
            statsGrid.innerHTML = '';
            checks.forEach(check => {
                const card = document.createElement('div');
                card.className = `stat-card ${check.status}`;
                card.innerHTML = `
                    <div class="stat-header">
                        <span class="stat-title">${check.title}</span>
                        <span class="badge ${check.status}">${check.status}</span>
                    </div>
                    <div class="stat-desc">${check.text}</div>
                    ${check.detail ? `<div class="stat-detail">${escapeHtml(check.detail)}</div>` : ''}
                `;
                statsGrid.appendChild(card);
            });

            // Scroll to results
            resultsContainer.scrollIntoView({ behavior: 'smooth' });
        }

        function escapeHtml(str) {
            return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
        }
